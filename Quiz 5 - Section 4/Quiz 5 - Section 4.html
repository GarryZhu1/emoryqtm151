(c) Compute growth rates

  • Calculate a new column "growth_cb" which computes the percentage daily

growth rate of "cb_to_usd".

Hint: Use .diff() and .shift(1) as in Lecture 22

In [ ]:
# Write your own code

dataset["growth_cb1"] = dataset["cb_to_usd"].diff()
dataset["growth_cb2"]    = dataset["cb_to_usd"].shift(1)
dataset["growth_cb"] = dataset["growth_cb1"]* 100/dataset["growth_cb2"]

(d) Flag subset of data and plot

  • Create a boolean colum (True/False) "bool_cond" which checks whether

"growth_cb" had a value greater than 15 or less than -15

  • Generate a line plot with the "growth_cb" series, label the axes and the title
  • Use the command plt.fill_between() as in Lecture 22 to create a shaded region

# Write your own code
dataset["bool_cond"] = (dataset["growth_cb"]>15)|(dataset["growth_cb"]<-15)

plt.plot("date", "growth_cb", data = dataset)
plt.xlabel("Time")
plt.ylabel("Daily percentage change ")
plt.title("Periods")

vec_y = dataset["growth_cb"]
plt.fill_between(x= dataset["date"],
                 y1 = vec_y.min(),
                 y2 = vec_y.max(),
                 where = dataset["bool_cond"],
                 alpha = 0.2,color = "red")

plt.show()

month_config = pd.Grouper(key='date', freq='m')

agg_data = (dataset 
               .groupby(month_config)
               .agg({'cb_to_usd': 'std'})
               .reset_index())
